```
function getSequence(n) {
  let sequence = [1,2];
  let nextValue = 0;
  for (let index=2; index<n; index++){
    nextValue = sequence[index-1] + sequence[index-2];
    sequence.push(nextValue);
  }

  return sequence;
}

console.log(getSequence(3));
console.log(getSequence(4));
console.log(getSequence(5));
```
