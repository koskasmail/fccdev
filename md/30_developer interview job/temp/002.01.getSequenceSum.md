function getSequenceSum(n) {
  let sequence = [1,2];
  let nextValue = 0;
  let sum = 1 + 2;

  if (n === 1) 
     return n;

  for (let index=2; index<n; index++){
    nextValue = sequence[index-1] + sequence[index-2];
    sequence.push(nextValue);
    sum += nextValue;
  }

  console.log(sequence); 
  return sum;
}
