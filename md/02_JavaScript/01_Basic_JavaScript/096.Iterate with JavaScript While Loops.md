### [ Task ]
  1. Add the numbers 5 through 0 (inclusive) in descending order to `myArray` using a `while` loop.


### [ before ]

```sh
// Setup
const myArray = [];

// Only change code below this line
```

### [ solution ]

```sh
// Setup
const myArray = [];

// Only change code below this line

let index = 0;
while (index < 5) {
    myArray.push(5 - index);
    index++;
}

console.log(myArray);

index = 0;
while (index < 5) {
    console.log(index+1 + '> ' + myArray[index]);
    index++;
}
```
